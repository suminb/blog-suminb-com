	LCS(X, Y)
		for(i=1; i<=m; i++) L[i, 0] = 0
		for(j=0; j<=n; j++) L[0, j] = 0
		for(i=1; i<=m; i++)
			for(j=1; j<=n; j++)
				if(x[i] == y[j])
					L[i, j] = 1 + L[i-1, j-1]
				else if(L[i-1, j] >= L[i, j-1])
					L[i, j] = L[i-1, j]
				else
					L[i, j] = L[i, j-1]
